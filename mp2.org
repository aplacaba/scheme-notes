(define (annual-po-mp2 yr pct initial-amt)
  (define total (+ initial-amt
                   (* pct initial-amt)))
  (if (= yr 0)
      0
      (+ total
         (annual-po-mp2
           (- yr 1)
           pct
           initial-amt))))

(define (compound-mp2 yr pct initial-amt)
  (define total (+ initial-amt
                   (* pct initial-amt)))

  (if (= yr 0)
      0
      (+ total
         (compound-mp2
           (- yr 1)
           pct
           total))))
